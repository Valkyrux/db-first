CARS
    ID - INT, PRIMARY_KEY, AUTOINCREMENT, NOT_NULL, UNIQUE;
    BRAND - VARCHAR(100), NOT_NULL;
    MODEL - VARCHAR(100), NOT_NULL;
    MODEL_YEAR - YEAR;
    FUEL - VARCHAR(20), NOT_NULL;
    NEW - BOOL, NOT_NULL;
    PURCHASE_YEAR - YEAR; //solo se NEW == FALSE
    DISTANCE - INT;
    AIR_CONDITIONER - BOOL;
    DESCRIPTION - TEXT;
    WAREHOUSE_NUMBER - TINYINT();
    AVIABLE_FOR_PURCHASE - BOOL, NOT_NULL;
    PRICE - FLOAT(8,2);
